CREATE TABLE [dbo].[CARRO]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY (1,1),
	[MODELO] VARCHAR(100) NOT NULL,
	[ANO] VARCHAR(50) NOT NULL,
	[VENDAS] INT NOT NULL,
	[DATAVENDA] DATE NOT NULL
)

SELECT 
	MODELO,
	ANO AS 'ANO DO MODELO',
	YEAR (DATAVENDA) AS 'ANO DA VENDA',
	SUM(VENDAS)/
	COUNT(DATAVENDA) AS 'VENDIDOS NO ANO'
	FROM CARRO
GROUP BY MODELO, ANO, DATAVENDA
ORDER BY DATAVENDA DESC
